function youtube-api.authenticate (options)

  • description and source-code
    authenticate = function (options) {
    if (!options) {
        config.auth = undefined;
        return;
    }

    var authObj = null;
    switch (options.type) {
        case "oauth":
            authObj = new Google.auth.OAuth2(options.client_id, options.client_secret, options.redirect_url);
            authObj.setCredentials({
                access_token: options.access_token || options.token,
                refresh_token: options.refresh_token
            });
            break;
        case "key":
            authObj = options.key;
            break;
    }

    Google.options({ auth: authObj });
    config.auth = options;

    return authObj;
}
